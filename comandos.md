# comandos de git en clases
- git init
- git add tech.md
- git commit -m "Tecnologías que se utilizan en el desarrollo de software"
- git log
- git log --oneline
- git status
- git branch desarrollo Crea nuevas ramas
- git switch desarrollo Cambia de rama
- git checkout master permite actualizar el archivo
- git merge desarrollo permite actualizar la rama desarrollo con master
- git remote add origin https://github.com/RobertoGarcia85/Formacion-docente-modulo-3-kodigo.git
- git branch -M main
- git push -u origin main
- git clone https://github.com/ferjoaguilar/mi-repo-de-la-formacion-docente.git
- git pull para traer algun archivo de un repositorio
- git reset --hard id del commit (para borrar los ultimos cambios)
- git reflog (para ver el historial de cambios)
- git reset --hard ultimo id del commit (para borrar los ultimos cambios)
